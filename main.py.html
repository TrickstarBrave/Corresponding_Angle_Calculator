angle = input("Welcome to the corrensponding angle calculator :D\n==================================================\nPlease enter your angle and we'll handle the rest ")
if int(angle) < 90:
    print(angle)
elif int(angle) < 180:
    print(180-int(angle))
elif int(angle) <270:
    print(int(angle)-180)
elif int(angle) <360:
    print(360 - int(angle))
